Plot

pub struct MyPlot {
    show_axes: [bool; 2],
    allow_drag: bool,
    allow_zoom: bool,
    center_x_axis: bool,
    center_y_axis: bool,
    width: f32,
    height: f32,
}

impl Default for MyPlot {
    fn default() -> Self {
        Self {
            show_axes: [true, true],
            allow_drag: true,
            allow_zoom: true,
            center_x_axis: false,
            center_y_axis: false,
            width: 300.0,
            height: 200.0,
        }
    }
}
    fn example_plot(&self, ui: &mut egui::Ui) -> egui::Response {
        use egui::plot::{Line, Value, Values};
        let n = 128;
        fn sigmoid(x: f64) -> f64 {
            -1.0 + 2.0 / (1.0 + f64::exp(-x))
        }
        let line = Line::new(Values::from_values_iter((0..=n).map(|i| {
            use std::f64::consts::TAU;
            let x = egui::remap(i as f64, 0.0..=n as f64, -TAU..=TAU);
            Value::new(x, sigmoid(x))
        })));

        let s = MyPlot::default();
        let response = egui::plot::Plot::new("example_plot")
            .show_axes(s.show_axes)
            .allow_drag(s.allow_drag)
            .allow_zoom(s.allow_zoom)
            .center_x_axis(s.center_x_axis)
            .center_x_axis(s.center_y_axis)
            .width(s.width)
            .height(s.height)
            .data_aspect(1.0)
            .show(ui, |plot_ui| plot_ui.line(line))
            .response;
        self.add_context_menu(response)
    }

draw something

let TAU = 30.0;
let size = Vec2::splat(16.0);
let (response, painter) = ui.allocate_painter(size, egui::Sense::hover());
let rect = response.rect;
let c = rect.center();
let r = rect.width() / 2.0 - 1.0;
let color = Color32::from_gray(128);
let stroke = egui::Stroke::new(1.0, color);
painter.circle_stroke(c, r, stroke);
painter.line_segment([c - egui::vec2(0.0, r), c + egui::vec2(0.0, r)], stroke);
painter.line_segment([c, c + r * Vec2::angled(TAU * 1.0 / 8.0)], stroke);
painter.line_segment([c, c + r * Vec2::angled(TAU * 3.0 / 8.0)], stroke);
